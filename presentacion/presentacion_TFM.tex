%--------------------------- 3 -------------------------------------------
\setslidetag{Contexto}
\begin{frame}{RSA dentro de la confianza web}
\slidebody{
  \begin{columns}[T,totalwidth=\linewidth]
    \column{0.50\linewidth}
      \includegraphics[height=0.64\textheight]{tls_handshake.png}
    \column{0.46\linewidth}
      \begin{block}{Punto de partida}
        La Web confia en certificados digitales que contienen claves publicas. Una parte relevante de ese ecosistema usa RSA.
      \end{block}
      \vspace{0.25cm}
      \begin{itemize}
        \item En RSA, la clave publica contiene un modulo \(N=p\cdot q\).
        \item La seguridad practica depende de que \(p\) y \(q\) sean secretos, grandes e independientes.
        \item Un certificado publicado permite auditar el modulo publico, no la clave privada.
      \end{itemize}
  \end{columns}
}
\end{frame}

%--------------------------- 4 -------------------------------------------
\setslidetag{Problema}
\begin{frame}{La vulnerabilidad: compartir un primo}
\slidebody{
  \begin{columns}[c,totalwidth=\linewidth]
    \column{0.52\linewidth}
      \includegraphics[width=\linewidth]{fig_gcd_compartido.png}
    \column{0.44\linewidth}
      \[
        N_1=pq_1,\qquad N_2=pq_2
      \]
      \[
        \gcd(N_1,N_2)=p
      \]
      \vspace{0.15cm}
      \begin{block}{Consecuencia}
        Recuperar \(p\) factoriza ambos modulos. No se rompe RSA por factorizar un entero dificil, sino por explotar un fallo de generacion.
      \end{block}
      \vspace{0.15cm}
      {\small En condiciones ideales, esta coincidencia deberia ser practicamente imposible.}
  \end{columns}
}
\end{frame}

\begin{frame}{El cuello de botella: comparar todos los pares}
\slidebody{
  \begin{columns}[c,totalwidth=\linewidth]
    \column{0.45\linewidth}
      \smallmetric{3.7cm}{\(\binom{M}{2}\)}{comparaciones ingenuas}\\[0.22cm]

      \begin{itemize}\small
        \item El metodo clasico usa \textit{product tree} y \textit{remainder tree}.
        \item El coste practico se concentra en enteros intermedios muy grandes.
        \item El objetivo es conservar la idea batch, reduciendo trabajo y memoria.
      \end{itemize}
  \end{columns}

%--------------------------- 9 -------------------------------------------
\setslidetag{Algoritmo}
\begin{frame}{Binary Tree Batch GCD}
\slidebody{
  \begin{columns}[c,totalwidth=\linewidth]
    \column{0.54\linewidth}
      \includegraphics[width=\linewidth]{fig_binary_tree.png}
    \column{0.42\linewidth}
      \begin{enumerate}
        \item Construye un arbol de productos.
        \item Calcula \(\gcd\) entre hermanos durante la construccion.
        \item Agrega los \(g_j\) no triviales en \(B=\prod g_j\).
        \item Enumera \(\gcd(N_i,B)\) para recuperar factores.
      \end{enumerate}
      \vspace{0.2cm}
      \begin{block}{Idea clave}
        Evita construir explicitamente el \textit{remainder tree} del enfoque clasico.
      \end{block}
  \end{columns}
}
\end{frame}
